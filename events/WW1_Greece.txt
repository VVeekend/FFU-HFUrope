	###Schism begins
	id = ww1_greece.15

	desc = ww1_greece.15.d
	picture = GFX_report_event_gre_political_upheaval
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		is_in_faction = no
	}
	option = {
		name = ww1_greece.15.a
		set_country_flag = national_schism
		every_owned_state = {
			limit = {
				OR = {
					state = 731
					state = 184
					state = 757
					state = 826
					state = 345
					state = 758
					state = 757
					state = 200
					state = 187
				}
			}
			set_demilitarized_zone = yes
		}
		if = {
			#North Epirus
			limit = {
				owns_state = 803
			}
			if = {
				limit = {
					country_exists = ALB
				}
				if = {
					limit = {
						ALB = {
							has_capitulated = yes
						}
					}	
					ALB = {
						set_state_owner = 803
					}
					if = {
						limit = {
							ITA = {
								has_war_together_with = ALB
							}
						}
						803 = {	set_state_controller_to = ITA }
						else = {
							every_country = {
								limit = {
									has_war_with = ALB
									any_state = {
										is_controlled_by = PREV
										is_owned_by = ALB
									
									}
								}
								803 = {	set_state_controller_to = PREV }
							}						
						}
					}
					else = {
						803 = {	transfer_state_to = ALB }
					}
				}
				else = {
					803 = {
						set_demilitarized_zone = yes
					}
				}
			}
		}
		add_ideas = GRE_idea_national_schism
		add_ideas = GRE_idea_dual_power
		ai_chance = {
			factor = 1
		}
	}
}
