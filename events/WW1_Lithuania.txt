
add_namespace = ww1_lithuania

# Remove Kingdom of Lithuania
####################
country_event = {
	id = ww1_lithuania.1
	trigger = {
		tag = LIT
		has_cosmetic_tag = LIT_GER
		OR = {
			AND = {
				is_subject_of = GER
				GER = {
					has_country_flag = great_war_loser
				}
			}
			OR = {
				AND = {
					is_subject = no
					has_country_flag = great_war_loser
				}
			}	
		}
		NOT = {
			has_global_flag = versallies_treaty
		}
	}
	mean_time_to_happen = {
		days = 5
	}
	fire_only_once = yes
	hidden = yes
	immediate = {
		drop_cosmetic_tag = yes
		set_party_name = {
			ideology = neutrality
			long_name = LIT_neutrality_party_long
			name = LIT_neutrality_party
		}
		LIT_antanas_smetona = {
			promote_character = authoritarian_democracy
		}		
		if = {
			limit = {
				is_subject_of = GER
			}
			GER = {
				end_puppet = LIT
			}
			leave_faction = yes
		}
	}
}

# Death of Mindaugas II
####################
country_event = {
	id = ww1_lithuania.2
	title = ww1_lithuania.2.t
	desc = ww1_lithuania.2.d
	trigger = {
		has_country_leader = {
			character = LIT_mindaugas_ii
			ruling_only = no
		}
		date > 1928.3.24
	}
	mean_time_to_happen = {
		days = 0
	}
	fire_only_once = yes
	option = {
		name = ww1_lithuania.2.a
		retire_character = LIT_mindaugas_ii
		LIT_vytautas_ii = {
			add_country_leader_role = {
				country_leader = {
					ideology = autocracy
					traits = { german_agent  }
					expire = "1935.1.1.1"
				}
			}
			promote_character = autocracy
		}
	}
}
